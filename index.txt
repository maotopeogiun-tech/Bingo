<script>

// ======================
// 중요
// ======================
// 반드시 본인 Supabase 값 넣기
// 그리고 Storage bucket을
// Public 으로 설정해야
// 이미지가 안깨짐
// ======================

const SUPABASE_URL =
'https://YOUR_PROJECT.supabase.co';

const SUPABASE_KEY =
'YOUR_ANON_KEY';

const supabaseClient =
supabase.createClient(
  SUPABASE_URL,
  SUPABASE_KEY
);

// ======================
// 라벨
// ======================

const LABELS = [
"거울샷","담임선생님과 함께","우리들의 야식샷","파워레인저샷","그림자샷",
"별 손가락 샷","스누피와 함께","축소샷","남학생 공항룩","비행기샷",
"헬멧착용샷","여학생 공항룩","단체샷","2만보 인증샷","리더의 무게샷",
"입 들어가샷","해양생물과 함께","별모양포즈","기념품 플렉스샷","오설록 포즈",
"바퀴벌레샷","남녀학생 함께","끌려다니기 샷","하트샷","돌하르방샷"
];

// ======================
// 방 생성
// ======================

function createRoom(name,password){

  return {
    name,
    password,
    bingo:0,
    cells:Array.from(
      {length:25},
      (_,i)=>({
        label:LABELS[i],
        image:null
      })
    )
  };
}

// ======================
// 기본 1반
// 비밀번호 20261
// ======================

let rooms = [
  createRoom('1반','20261')
];

let currentRoom = 0;

let pendingCell = null;

// ======================
// DOM
// ======================

const board =
document.getElementById('board');

const roomTabs =
document.getElementById('roomTabs');

const previewGrid =
document.getElementById('previewGrid');

const fileInput =
document.getElementById('fileInput');

// ======================
// 저장
// ======================

function saveData(){

  localStorage.setItem(
    'bango_rooms',
    JSON.stringify(rooms)
  );
}

function loadData(){

  const saved =
  localStorage.getItem(
    'bango_rooms'
  );

  if(saved){

    rooms =
    JSON.parse(saved);
  }
}

// ======================
// 점수
// ======================

function score(room){

  let filled =
  room.cells.filter(
    c=>c.image
  ).length;

  return filled * 2;
}

// ======================
// 방 렌더
// ======================

function renderRooms(){

  roomTabs.innerHTML = '';

  rooms.forEach((room,i)=>{

    const btn =
    document.createElement('div');

    btn.className =
    'room-btn' +
    (
      i===currentRoom
      ? ' active'
      : ''
    );

    btn.textContent =
    room.name;

    btn.onclick = ()=>{

      // 현재 방이면 바로 입장
      if(i===currentRoom){
        return;
      }

      const pw =
      prompt(
        `${room.name} 비밀번호 입력`
      );

      if(pw !== room.password){

        alert('비밀번호 틀림');
        return;
      }

      currentRoom = i;

      renderAll();
    };

    roomTabs.appendChild(btn);
  });
}

// ======================
// 빙고판
// ======================

function renderBoard(){

  board.innerHTML = '';

  const room =
  rooms[currentRoom];

  room.cells.forEach((cell,i)=>{

    const div =
    document.createElement('div');

    div.className = 'cell';

    // ======================
    // 이미지 있으면 표시
    // ======================

    if(cell.image){

      const img =
      document.createElement('img');

      img.src = cell.image;

      // 이미지 깨졌을때
      img.onerror = ()=>{

        div.innerHTML =
        `<div class="empty">
          이미지 오류
        </div>`;
      };

      div.appendChild(img);

      const label =
      document.createElement('div');

      label.className =
      'label';

      label.textContent =
      cell.label;

      div.appendChild(label);

    }else{

      div.innerHTML =
      `<div class="empty">
        ${cell.label}
      </div>`;
    }

    div.onclick = ()=>{

      pendingCell = i;

      fileInput.click();
    };

    board.appendChild(div);
  });

  document.getElementById(
    'score'
  ).textContent =
  score(room);

  renderPreview();
}

// ======================
// 미리보기
// ======================

function renderPreview(){

  previewGrid.innerHTML = '';

  rooms[currentRoom]
  .cells
  .forEach(cell=>{

    if(!cell.image) return;

    const card =
    document.createElement('div');

    card.className =
    'preview-card';

    card.innerHTML = `
      <img src="${cell.image}">
      <div>${cell.label}</div>
    `;

    previewGrid.appendChild(card);
  });
}

// ======================
// 이미지 업로드
// ======================

async function uploadImage(file){

  // ======================
  // base64 저장
  // 이미지 절대 안깨짐
  // ======================

  return new Promise(resolve=>{

    const reader =
    new FileReader();

    reader.onload = e=>{

      resolve(e.target.result);
    };

    reader.readAsDataURL(file);
  });
}

// ======================
// 파일 선택
// ======================

fileInput.addEventListener(
  'change',
  async e=>{

    const file =
    e.target.files[0];

    if(!file) return;

    const image =
    await uploadImage(file);

    rooms[currentRoom]
    .cells[pendingCell]
    .image = image;

    saveData();

    renderBoard();
  }
);

// ======================
// 방 추가
// ======================

document
.getElementById(
  'addRoomBtn'
)
.onclick = ()=>{

  const name =
  prompt('방 이름');

  if(!name) return;

  const password =
  prompt('방 비밀번호');

  if(!password) return;

  rooms.push(
    createRoom(
      name,
      password
    )
  );

  saveData();

  renderAll();
};

// ======================
// 전체 렌더
// ======================

function renderAll(){

  renderRooms();

  renderBoard();
}

// ======================
// 시작
// ======================

loadData();

renderAll();
